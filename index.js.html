constructor(request, options) {
    // Debug
    log.debug('header', 'debug',
      `class: ${Header.name}`, 'process: constructor');
    log.debug(`class: ${Header.name}`, `request exists: ${ !(!request) }`);
    log.info(`options exists: ${!(!request) && !_.isEmpty(options) }`);
    // Set Header
    this.header = accept(this.request, this.options);
    // Set Locale
    this.locale = this.getLocale();
  }

export default () => {
'use strict';
return {
  main: function(req) {
    var options = this.options.header;
    this.header = new Header(req, options);
  },
  package: _.merge({
    type: 'header'
  }, require('../package')),
  defaults: require('../defaults')
};
};
